高精度

3 min read

简介

高精度是一套支持高位数的运算系统，例如求两个 114514 位数的和。

输入

先输入字符串，再把每一位取出，存入数组。

void init(int a[]) {
    memset(a, 0, sizeof a); // 清空数组 a
    string s; cin >> s;
    a[0] = s.length(); // a[0] 用于储存数的长度
    for(int i = 1; i <= a[0]; i ++)
        a[i] = s[a[0] - i] - '0'; // 将字符串倒序储存，便于按位计算
}

也可以逐个读入字符。

void init(int a[]) {
    memset(a, 0, sizeof a);
    int i = 0; char s = getchar();
    while(s < '0' || s > '9') s = getchar();
    while(s >= '0' && s <= '9')
        a[++ i] = s – '0', s = getchar();
    a[0] = i;
}

比较

先比较位数，若位数相同则从高位到低位逐位比较。

int cmp(int a[], int b[]) { // a > b 则返回 1，a < b 则返回 -1，否则返回 0
    if(a[0] != b[0]) return a[0] > b[0] ? 1 : -1;
    for(int i = a[0]; i >= 1; i --)
        if(a[i] != b[i]) return a[i] > b[i] ? 1 : -1;
    return 0; 
}

高精度加法

从个位开始，将两个数对应位置的数字相加，超过 $10$ 要进位。

void plus(int a[], int b[], int c[]) { // 将 a + b 的结果存入 c
    memset(c, 0, sizeof c);
    c[0] = max(a[0], b[0]);
    for(int i = 1; i <= c[0]; i ++) {
        c[i] += a[i] + b[i];
        if(c[i] >= 10) // 进 1
            c[i + 1] += 1, c[i] %= 10;
    }
    while(c[c[0] + 1] != 0) c[0] ++; // 修正位数
}

高精度减法

从个位开始，将两个数对应位置的数字相减，小于 $0$ 要借位。

void minus(int a[], int b[], int c[]) { // 将 a - b 的结果存入 c
    memset(c, 0, sizeof c);
    c[0] = max(a[0], b[0]);
    for(int i = 1; i <= c[0]; i ++) {
        if(a[i] < b[i]) // 借位
            a[i + 1] --, a[i] += 10;
        c[i] = a[i] - b[i];
    }
    while(c[c[0]] == 0 && c[0] >= 1) c[0] --; // 修正位数
}

高精度乘法

高精 × 低精

从个位开始，将高精度数的每一位乘以低精度数，超过 $10$ 要进位。

void mult(int a[], int b, int c[]) { // 将 a × b 的结果存入 c
    memset(c, 0, sizeof c);
    c[0] = a[0];
    for(int i = 1; i <= a[0]; i ++) {
        c[i] += a[i] * b;
        if(c[i] >= 10)
            c[i + 1]  += c[i] / 10, c[i] /= 10;
    }
    while(c[c[0] + 1]) // 处理多余的进位
        c[c[0] + 1] += c[c[0]] / 10, c[c[0]] /= 10, c[0] ++;
}

高精 × 高精

做竖式乘法时，$a$ 的第 $i$ 位乘 $b$ 的第 $j$ 位最终会加在答案的第 $i+j-1$ 位上。

计算结束后要去掉最高位多余的 $0$。

void mult(int a[], int b[], int c[]) { // 将 a × b 的结果存入 c
    memset(c, 0, sizeof c);
    c[0] = a[0] + b[0]; 
    int d; // 用于存放进位
    for(int i = 1; i <= a[0]; i ++) {
        d = 0;
        for(int j = 1; j <= b[0]; j ++) {
            c[i + j - 1] += a[i] * b[j] + d;
            d = c[i + j - 1] / 10, c[i + j - 1] %= 10; 
        }
        c[i + b[0]] =  d; // 多余的进位
    }
    while(c[c[0]] == 0 && c[0] > 1) c[0] --; // 去掉最高位多余的 0
}

高精度除法

做竖式除法时，将除数的最高位和被除数对齐，逐位「试商」，如下图（空位已用 $0$ 填补）：

高精度除法步骤：

  1. 将除数的最高位和被除数对齐；

  2. 将被除数一直减去除数，直到被除数 < 除数，记录减的次数；

  3. 重复上两步，直到除到个位；

  4. 按顺序输出减的次数，得到整个式子的商。最后的被除数是余数。

计算结束后要去掉最高位多余的 $0$。

样例程序的 dev() 函数执行结束后，数组 a 存储的数变为 a÷b 的余数。

void shift(int a[], int d) { // 在 a 的后面补 d 个 0（若 d < 0 则去掉|d|个 0）
    int b[100001] = {};
    for(int i = a[0] + d; i >= max(0, 1 + d); i --)
        b[i] =  a[i - d];
    a[0] += d;
    for(int i = 1; i <= max(a[0], a[0] - d); i ++)
        a[i] = b[i];
}

void dev(int a[], int b[], int c[]) { // 将 a ÷ b 的结果存入 c
    memset(c, 0, sizeof c);
    if(b[0] == 1 && b[1] == 0) return; // 除数为 0 时无意义
    if(cmp(a, b) == -1) { // 被除数 < 除数时，商为 0
        c[0] = 1, c[1] = 0; return;
    }
    c[0] = a[0] - b[0] + 1;
    shift(b, a[0] - b[0]); // 调整 b 的位数，使 b 和 a 对齐
    for(int i = c[0]; i >= 1; i --) {
        while(cmp(a, b) >= 0) { // 用高精度减法模拟「试商」
            c[i] ++;
            for(int j = 1; j <= a[0]; j ++) {
                if(a[j] < b[j])
                    a[j + 1] --, a[j] += 10;
                a[j] -= b[j];
            }
            while(a[a[0]] == 0 && a[0] > 1) a[0] --; // 去掉最高位多余的 0
        }
        shift(b, -1); // 将 b 减小一位，和新的被除数对齐
    }
    while(c[c[0]] == 0 && c[0] > 1) c[0]--; // 去掉最高位多余的 0
}

输出

数据是倒序储存的，故倒序输出：

void print(int a[]) {
    for(int i = a[0]; i >= 1; i --)
        printf("%d", a[i]);
    printf("\n");
}
