    STL

    简介$

    STL（Standard Template Library 标准模板库）是 C++ 开发者提供的通用模板类和函数库。

    Bitset$

    函数 功能
    #include <bitset> 头文件
    bitset<n> b 创建 n 位二进制数 bitset
    bitset<n> b(val) 将 bitset 初始化为 val
    b.count() 返回 bitset 有几位为 1
    b.flip() 将 bitset 取反
    b.flip(pos) 将 bitset 的第 pos 位取反（pos 从 0 记起）
    b.reset() 将 bitset 每一位都赋值为 0
    b.reset(pos) 将 bitset 的第 pos 位赋值为 0
    b.set() 将 bitset 每一位都赋值为 1
    b.set(pos) 将 bitset 的第 pos 位赋值为 1
    b.set(pos, x) 将 bitset 的第 pos 位赋值为 x
    b.size() 返回 bitset 位数

    Queue$

    函数 功能
    #include <queue> 头文件
    queue<type> q 创建 type 类的 queue
    q = queue<type>() 清空 queue
    q.back() 返回 queue 末尾元素
    q.empty() 返回 queue 是否为空
    q.front() 返回 queue 头部元素
    q.push(x) 将 x 插入 queue 的末尾
    q.pop() 删除 queue 尾部元素
    q.size() 返回 queue 元素个数

    Deque$

    函数 功能
    #include <deque> 头文件
    deque<type> q 创建 type 类的 deque
    q = deque<type>() 清空 deque
    q.pop_back() 删除 deque 尾部元素
    q.pop_front() 删除 deque 头部元素
    其它操作和 Queue 相同

    Priority Queue$

    函数 功能
    #include <queue> 头文件
    priority_queue<type> q 创建 type 类的 priority_queue（默认为降序）
    priority_queue<type, vector<type>,
    greater<type> > q    		 创建升序 priority_queue
    priority_queue<type, vector<type>,
    less<type> > q    		 创建降序 priority_queue
    q.empty() 返回 priority_queue 是否为空
    q.push(x) 将 x 插入 priority_queue 并排序
    q.pop() 删除 priority_queue 顶部元素
    q.size() 返回 priority_queue 元素个数
    q.top() 返回 priority_queue 顶部元素

    Stack$

    函数 功能
    #include <stack> 头文件
    stack<type> s 创建 type 类的 stack
    s = stack<type>() 清空 stack
    s.empty() 返回 stack 是否为空
    s.push(x) 将 x 加入 stack 顶部
    s.pop() 删除 stack 顶部元素
    s.size() 返回 stack 元素个数
    s.top() 返回 stack 顶部元素

    Vector$

    函数 功能
    #include <vector> 头文件
    vector<type> vec 创建 type 类的 vector
    vector<type> vec(n) 创建包含 n 个 type 类元素的 vector
    vector<type> vec(n, val) 创建包含 n 个元素的 vector，且元素初始化为 val
    vec[pos] 返回 vector 第 pos 个元素的引用（pos 从 0 记起）
    vec.at(pos) 返回 vector 第 pos 个元素的值
    vec.back() 返回 vector 末尾元素
    vec.clear() 清空 vector
    vec.empty() 返回 vector 是否为空
    vec.erase(pos) 删除 pos 位置的元素
    vec.erase(beg, end) 删除 [beg, end) 区间的元素
    vec.front() 返回 vector 头部元素
    vec.insert(pos, x) 在 pos 位置插入 x
    vec.insert(pos, n, x) 在 pos 位置插入 n 个 x
    vec.push_back(x) 在 vector 末尾插入 x
    vec.pop_back() 删除 vector 末尾元素
    vec.size() 返回 vector 元素个数
