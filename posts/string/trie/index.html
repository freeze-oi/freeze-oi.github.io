    Trie树

    Code
    int trie[][], tot;
bool end[];

void insert(string str) { // 往 trie 树中插入字符串 str
    int p = 0;
    for(int i = 0; i < str.size(); i ++) {
        int ch = str.at(i);
        if(!trie[p][ch]) trie[p][ch] = ++ tot;
        p = trie[p][ch];
    }
    end[p] = true;
}

bool search(string str) { // 查询 trie 树中是否有字符串 str
    int p = 0;
    for(int i = 0; i < str.size(); i ++) {
        int ch = str.at(i);
        if(!trie[p][ch]) return false;
        p = trie[p][ch];
    }
    return end[p];
}
